# Функция для вычисления факториала
def factorial(n):
    if n < 0:
        return "Ошибка: факториал отрицательного числа не существует"
    result = 1
    for i in range(2, n + 1):
        result *= i
    return result

# Основной код
number = int(input("Введите число: "))
print(f"Факториал числа {number} равен {factorial(number)}")

# Формула Герона для вычисления площади треугольника
def triangle_area(a, b, c):
    # Полупериметр
    p = (a + b + c) / 2
    # Площадь по формуле Герона
    area = (p * (p - a) * (p - b) * (p - c)) ** 0.5
    return area

# Основной код
a = float(input("Введите сторону a: "))
b = float(input("Введите сторону b: "))
c = float(input("Введите сторону c: "))

if a + b > c and a + c > b and b + c > a:
    print(f"Площадь треугольника: {triangle_area(a, b, c)}")
else:
    print("Треугольник с такими сторонами не существует")

# Функция для выполнения операций
def calculator(num1, num2, operation):
    if operation == '+':
        return num1 + num2
    elif operation == '-':
        return num1 - num2
    elif operation == '*':
        return num1 * num2
    elif operation == '/':
        if num2 == 0:
            return "Ошибка: деление на ноль"
        return num1 / num2
    else:
        return "Ошибка: неверная операция"

# Основной код
print("Доступные операции: +, -, *, /")
num1 = float(input("Введите первое число: "))
num2 = float(input("Введите второе число: "))
operation = input("Введите операцию: ")

result = calculator(num1, num2, operation)
print(f"Результат: {result}")

# Функция для сравнения чисел
def compare_numbers(num1, num2):
    if num1 > num2:
        return f"{num1} больше {num2}"
    elif num1 < num2:
        return f"{num1} меньше {num2}"
    else:
        return f"{num1} равно {num2}"

# Основной код
num1 = int(input("Введите первое число: "))
num2 = int(input("Введите второе число: "))

print(compare_numbers(num1, num2))
