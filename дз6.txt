import java.util.*;

// Класс Student
public class Student {
    private String name;
    private String group;
    private int course;
    private Map<String, Double> grades; // предмет - оценка

    // Конструктор
    public Student(String name, String group, int course) {
        this.name = name;
        this.group = group;
        this.course = course;
        this.grades = new HashMap<>();
    }

    // Методы для работы с оценками
    public void addGrade(String subject, double grade) {
        grades.put(subject, grade);
    }

    public double getAverageGrade() {
        if (grades.isEmpty()) return 0.0;
        double sum = 0.0;
        for (double grade : grades.values()) {
            sum += grade;
        }
        return sum / grades.size();
    }

    // Метод перевода на следующий курс
    public void promote() {
        course++;
    }

    // Геттеры
    public String getName() {
        return name;
    }

    public int getCourse() {
        return course;
    }

    // Метод для удаления неуспевающих студентов
    public static void removeUnsuccessful(Set<Student> students) {
        Iterator<Student> iterator = students.iterator();
        while (iterator.hasNext()) {
            Student student = iterator.next();
            if (student.getAverageGrade() < 3.0) {
                iterator.remove();
            }
        }
    }

    // Метод для перевода успешных студентов
    public static void promoteSuccessful(Set<Student> students) {
        for (Student student : students) {
            if (student.getAverageGrade() >= 3.0) {
                student.promote();
            }
        }
    }

    // Метод для печати студентов определенного курса
    public static void printStudents(Set<Student> students, int course) {
        for (Student student : students) {
            if (student.getCourse() == course) {
                System.out.println(student.getName());
            }
        }
    }
}

// Пример использования
public class Main {
    public static void main(String[] args) {
        Set<Student> students = new HashSet<>();
        
        // Создаем студентов
        Student student1 = new Student("Иван", "М-1", 1);
        Student student2 = new Student("Анна", "М-1", 1);
        
        // Добавляем оценки
        student1.addGrade("Математика", 4.0);
        student1.addGrade("Физика", 3.5);
        
        student2.addGrade("Математика", 2.0);
        student2.addGrade("Физика", 2.5);
        
        // Добавляем в коллекцию
        students.add(student1);
        students.add(student2);
        
        // Используем методы
        Student.printStudents(students, 1);
        Student.removeUnsuccessful(students);
        Student.promoteSuccessful(students);
    }
}
